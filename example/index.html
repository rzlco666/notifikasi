📝 Usage Example

// PHP Native
use Rzlco\Notifikasi;
use Rzlco\Storage\SessionStorage;

$notifikasi = new Notifikasi(new SessionStorage());
$notifikasi->success('Success!', 'Data berhasil disimpan');
echo $notifikasi->render();

// Laravel
use Rzlco\Facades\Notifikasi;

Notifikasi::success('Success!', 'Data berhasil disimpan');
// Di Blade: @notifikasi
